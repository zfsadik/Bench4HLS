The following adder-subtractor with zero flag doesn't work. Fix the bug(s).

#include <ap_int.h>

void TopModule(bool do_sub, ap_uint<8> a, ap_uint<8> b, ap_uint<8> &out, bool &result_is_zero) {
#pragma HLS INLINE

    if (do_sub == 0) {
        out = a + b;
    } else {
        out = a - b;
    }

    if (~out) {
        result_is_zero = 1;
    } else {
        result_is_zero = 0;
    }
}

The top-level function should have the following prototype:
void TopModule(bool do_sub, ap_uint<8> a, ap_uint<8> b, ap_uint<8> &out, bool &result_is_zero)
